OPÇÃO A (recomendada, reversível): habilitar sudo NOPASSWD para git/docker/compose
  USER_SSH="deploy"
  cat > /etc/sudoers.d/techno-os <<SUDOEOF
${USER_SSH} ALL=(ALL) NOPASSWD: /usr/bin/git, /usr/bin/docker, /usr/local/bin/docker-compose, /usr/bin/docker-compose
SUDOEOF
  chmod 440 /etc/sudoers.d/techno-os

OPÇÃO C (alternativa): transferir ownership do repo para o usuário SSH
  USER_SSH="deploy"
  GROUP_SSH="deploy"
  chown -R ${USER_SSH}:${GROUP_SSH} "/app/techno-os-backend"
